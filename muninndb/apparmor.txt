#include <tunables/global>

profile muninndb flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # Capabilities
  capability,
  file,
  signal (send) set=(kill,term,int,hup,cont),
  network,
  mount,

  # S6-Overlay
  /init ix,
  /bin/** ix,
  /usr/bin/** ix,
  /run/{s6,s6-rc*,service}/** ix,
  /package/** ix,
  /command/** ix,
  /etc/services.d/** rwix,
  /etc/cont-init.d/** rwix,
  /etc/cont-finish.d/** rwix,
  /run/{,**} rwk,
  /dev/tty rw,

  # Bashio
  /usr/lib/bashio/** ix,
  /tmp/** rwk,

  # MuninnDB binary and data
  /opt/muninndb/** ix,
  /data/** rw,

  # Access to options.json and other addon data
  /data/** rw,

  # SSL certificates
  /ssl/** r,

  # Shared volumes
  /share/** rw,
  /media/** rw,

  # DNS resolution
  /etc/hosts r,
  /etc/resolv.conf r,
  /etc/nsswitch.conf r,
  /etc/passwd r,
  /etc/group r,

  # Device access
  /dev/null rw,
  /dev/urandom r,
  /dev/random r,

  # Proc filesystem
  /proc/*/mountinfo r,
  /proc/*/stat r,
  /proc/*/status r,

  # MuninnDB specific paths
  owner /data/muninndb/** rwk,
}
