#include <tunables/global>

profile dockhand flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # Capabilities
  capability,
  file,
  signal (send) set=(kill,term,int,hup,cont),
  network,
  mount,

  # S6-Overlay
  /init ix,
  /bin/** ix,
  /usr/bin/** ix,
  /run/{s6,s6-rc*,service}/** ix,
  /package/** ix,
  /command/** ix,
  /etc/services.d/** rwix,
  /etc/cont-init.d/** rwix,
  /etc/cont-finish.d/** rwix,
  /run/{,**} rwk,
  /dev/tty rw,

  # Bashio
  /usr/lib/bashio/** ix,
  /tmp/** rwk,

  # Dockhand application and data
  /opt/dockhand/** ix,
  /opt/dockhand/** r,
  /data/** rw,

  # Node.js / Bun runtime
  /usr/bin/node ix,
  /usr/bin/bun ix,
  /usr/lib/node_modules/** r,

  # SSL certificates
  /ssl/** r,

  # Docker socket access - critical for Dockhand
  /var/run/docker.sock rw,
  /run/docker.sock rw,
  /{,var/}run/docker.sock rw,

  # Docker runtime access
  /sys/fs/cgroup/** r,
  /proc/sys/net/ipv4/ip_forward r,

  # Shared volumes
  /share/** rw,

  # DNS resolution
  /etc/hosts r,
  /etc/resolv.conf r,
  /etc/nsswitch.conf r,
  /etc/passwd r,
  /etc/group r,

  # Device access needed for Docker operations
  /dev/null rw,
  /dev/urandom r,
  /dev/random r,
  /dev/net/tun rw,

  # Proc filesystem
  /proc/*/mountinfo r,
  /proc/*/stat r,
  /proc/*/status r,

  # Dockhand specific paths
  owner /data/dockhand/** rwk,
  owner /data/stacks/** rwk,

  # Allow process tracing for container management
  ptrace (read,trace) peer=docker-default,
  ptrace (read,trace) peer=unconfined,

  # Git operations (for stack management)
  /usr/bin/git ix,
  /usr/libexec/git-core/** ix,
}
